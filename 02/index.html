Given an array with repeated elements/numbers, return an array of its index occurences.
Input = [1, 3, 1, 6, 1, 8, 2, 1, 0];
Search for element = 1;
Output = [0, 2, 4, 7];

Output Array: 



			<h3>Input Array: </h3>
			<p id="input"></p>
			<h3>Output Array: </h3>
			<p id="output"></p>

			<script>
				var inputArr = [1, 3, 1, 6, 1, 8, 2, 1, 0];
				document.getElementById("input").innerHTML = inputArr;

				function searchArray(array, search) {

					var outputArr = [];
					for(var i = 0; i < array.length; i++){
						if(array[i] == search){
							outputArr.push(i);
						}
					}
					return outputArr;
				}

				document.getElementById("output").innerHTML = searchArray(inputArr, 1);
			</script>
